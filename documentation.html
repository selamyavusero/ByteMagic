uintptr_t FindLuaSignature(HANDLE hProcess) {
    SYSTEM_INFO si;
    GetSystemInfo(&si);

    MEMORY_BASIC_INFORMATION mbi;
    uintptr_t addr = (uintptr_t)si.lpMinimumApplicationAddress;

    unsigned char magic[] = { 0x1B, 0x4C, 0x55, 0x41, 0x55 };
    size_t magicSize = sizeof(magic);

    while (addr < (uintptr_t)si.lpMaximumApplicationAddress) {
        if (VirtualQueryEx(hProcess, (LPCVOID)addr, &mbi, sizeof(mbi))) {
            if (mbi.State == MEM_COMMIT && (mbi.Protect & PAGE_READWRITE || mbi.Protect & PAGE_READONLY)) {
                std::vector buffer(mbi.RegionSize);
                SIZE_T bytesRead;

                if (ReadProcessMemory(hProcess, mbi.BaseAddress, buffer.data(), mbi.RegionSize, &bytesRead)) {
                    for (size_t i = 0; i < bytesRead - magicSize; i++) {
                        bool found = true;
                        for (size_t j = 0; j < magicSize; j++) {
                            if (buffer[i + j] != magic[j]) {
                                found = false;
                                break;
                            }
                        }
                        if (found) {
                            return (uintptr_t)mbi.BaseAddress + i;
                        }
                    }
                }
            }
            addr += mbi.RegionSize;
        }
        else {
            addr += 0x1000;
        }
    }
    return 0;
}

 void readbuff(HANDLE hProcess, uintptr_t address, size_t size) {
    std::vector buffer(size);
    SIZE_T bytesRead = 0;

    if (ReadProcessMemory(hProcess, (LPCVOID)address, buffer.data(), size, &bytesRead)) {
        printf("[+] Pulled %llu bytes. Running Deep Scan now \n\n", (unsigned long long)bytesRead);

        for (SIZE_T i = 0; i < bytesRead; i += 16) {
            std::vector chunk;
            for (int j = 0; j < 16 && (i + j) < bytesRead; j++) chunk.push_back(buffer[i + j]);

            // CHUNK-LEVEL BRUTE FORCE
            unsigned char bestKey = 0;
            int bestPrintable = -1;

            for (int k = 0; k < 256; k++) {
                int printableCount = 0;
                for (auto b : chunk) {
                    unsigned char decoded = b ^ k;
                    if (decoded >= 32 && decoded <= 126) printableCount++;
                    if (decoded == 0) printableCount += 2;
                }
                if (printableCount > bestPrintable) {
                    bestPrintable = printableCount;
                    bestKey = (unsigned char)k;
                }
            }

            printf("%04llX [%02X] ", (unsigned long long)i, bestKey);
            for (size_t j = 0; j < 16; j++) {
                if (j < chunk.size()) printf("%02x ", chunk[j] ^ bestKey);
                else printf("   ");
            }
            printf(" | ");
            for (size_t j = 0; j < chunk.size(); j++) {
                unsigned char ch = chunk[j] ^ bestKey;
                putchar((ch >= 32 && ch <= 126) ? ch : '.');
            }
            printf("\n");
        }
    }
    else {
        printf("[-] RPM Failed %lu\n", GetLastError());
    }
}

 void SuspendProcess(DWORD pid) {
    HANDLE snapshot = CreateToolhelp32Snapshot(TH32CS_SNAPTHREAD, 0);

    if (snapshot == INVALID_HANDLE_VALUE) {
        std::cerr << "[-] Failed to create snapshot. Error: " << GetLastError() << std::endl;
        return;
    }

    THREADENTRY32 te;
    te.dwSize = sizeof(te);

    if (Thread32First(snapshot, &te)) {
        do {
            if (te.th32OwnerProcessID == pid) {
                HANDLE hThread = OpenThread(THREAD_SUSPEND_RESUME, FALSE, te.th32ThreadID);
                if (hThread) {
                    DWORD result = SuspendThread(hThread);

                    //  0xFFFFFFFF is the thing we lowkey need more of in our lives.
                    if (result == (DWORD)-1) {
                        std::cerr << "[-] Could not suspend thread " << te.th32ThreadID
                            << ". Error: " << GetLastError() << std::endl;

                    CloseHandle(hThread);

}
